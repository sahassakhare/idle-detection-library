üíª Implementation Examples

Environment Configuration

// environment.ts export const environment = { production: false, idleConfig: { idleTimeout: 120000, // 2 minutes warningTimeout: 30000, // 30 seconds debug: true } };

Service Configuration Loading

// config.service.ts async loadConfig(): Promise<IdleConfig> { const configs = await this.http.get<AppConfig>( 'assets/config/idle-config.json' ).toPromise(); const env = this.getEnvironment(); return configs[env] || configs.production; }

Bootstrap Configuration

// main.ts bootstrapApplication(AppComponent, { providers: [ provideIdleOAuth(environment.idleConfig), // OR load from external JSON: { provide: 'IDLE_CONFIG_FACTORY', useFactory: (configService: ConfigService) => configService.getConfig(), deps: [ConfigService] } ] });

Runtime Configuration Updates

// component.ts updateTimeout(preset: string) { this.configService.applyPreset(preset); // Or custom configuration: this.configService.updateConfig({ idleTimeout: 30 * 60 * 1000, // 30 min warningTimeout: 5 * 60 * 1000 // 5 min }); }
